29 bool CursesWormsSimUIStrategy::processUserInput()
30 {
31  static const int numMicrosecondsInAMillisecond = 1000;
32 
33  showStatus();
34 
35  for (delayRemaining = slowness+1;
36  delayRemaining > 0;
37  delayRemaining -= delayQuantum)
38  {
39  char c = handleUserKeyPress(getch()); // no-delay
40  if (c == esc)
41  { // !!!! NOTE EARLY RETURN !!!!
42  return true;
43  }
44  if (isPaused)
45  {
46  delayRemaining += delayQuantum;
47  showStatus();
48  }
49  if (delayQuantum > 0)
50  {
51  usleep(numMicrosecondsInAMillisecond * delayQuantum);
52  }
53  }
54 
55  return false;
56 }

171 char CursesWormsSimUIStrategy::handleUserKeyPress(char c)
172 {
173  switch (c) {
174  case '+':
175  {
176  slowness -= std::min(slowness, 100); //< Arbitrary
177  break;
178  }
179  case '-':
180  {
181  slowness += 100; //< Arbitrary
182  break;
183  }
184  case 'f':
185  {
186  slowness = 0; // Let simulation run at maximum speed
187  break;
188  }
189  case 's':
190  {
191  /* TODO: highlight a worm */
192  break;
193  }
194  case 'k':
195  {
196  /* TODO: kill the highlighted worm */
197  break;
198  }
199  case 'w':

201  m_sim.createWorm();
202  break;
203  }
204  case ' ':

206  isPaused ^= 1;
207  break;
208  }
