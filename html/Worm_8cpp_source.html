79 void Worm::live(WormsSim &sim)
80 {
81  assert(!isAlive() || nullptr != m_typeInfo);
82  assert(!isAlive() || 1 < m_body.size());
83  assert(areAllSegmentsContiguous(WormsSim::getSingletonSim()));
84 
87  static const int numberDirections = 8;
88 
90  static const int dxa[numberDirections] = {
91  +0, +1, +1, +1, +0, -1, -1, -1 };
92 
94  static const int dya[numberDirections] = {
95  -1, -1, +0, +1, +1, +1, +0, -1 };
96 
97  // The number of possible direction indexes to choose
98  static const int distributionOfSelectableDirections = 16;
99 
104  static const int nextTurn[distributionOfSelectableDirections] = {
105  0, 0, 0, 0, 0, 0, 0, 0,
106  1, 1, 1, 7, 7, 7, 2, 6
107  };
108 
109  if (!isAlive())
110  { // !!!! EARLY EXIT !!!!
111  assert(m_body[0].c == ' ');
112  assert(areAllSegmentsContiguous(WormsSim::getSingletonSim()));
113  return;
114  }
115 
116  // Make each body segment move to the position of the next segment
117  auto headIt = getBody().end() - 1;
118  for(auto i = m_body.begin(); i < headIt; ++i)
119  {
120  i->x = (i + 1)->x;
121  i->y = (i + 1)->y;
122  }
123 
124  // Pick a movement direction relative to the current direction
125  // from the selectable directions
126  const int dir = (m_direction + nextTurn[WormsSim::getRandomModX(
127  distributionOfSelectableDirections)]) % numberDirections;
128 
129  m_direction = static_cast<Worm::direction>(dir);
130 
131  // Move head in chosen direction
132  getHead().y += dya[dir];
133  getHead().x += dxa[dir];
134 
135  // Wrap around edges of board
136  if (getHead().y < 0) getHead().y = sim.getHeight() - 1;
137  else if (getHead().y >= sim.getHeight()) getHead().y = 0;
138 
139  if (getHead().x < 0) getHead().x = sim.getWidth() - 1;
140  else if (getHead().x >= sim.getWidth()) getHead().x = 0;
141 
142  // Limit amount of food in stomach to capacity of body segments
143  m_stomach = std::min((int)getBody().size() * m_typeInfo->capacity,
144  m_stomach);
145 
146  // Consume some food
147  m_stomach -= 1;
148  if(isHungry())
149  {
150  m_typeInfo->eatFunction(*this, sim);
151  }
152 
153  updateStatusBasedOnStomach();
154 
155  assert(m_body[0].c == ' ');
156  assert(areAllSegmentsContiguous(WormsSim::getSingletonSim()));
157 }
